$$
\begin{aligned}
(p-1)! & \equiv \prod_{j=1}^{(p-1)/2} j(p-j) \equiv \prod_{j=1}^{(p-1)/2} (-j^2) \pmod{p} \\
& \equiv (-1)^{(p-1)/2} \left( \prod_{j=1}^{(p-1)/2} j \right)^2 \pmod{p},
\end{aligned}
$$

and apply Theorem 2.4.
